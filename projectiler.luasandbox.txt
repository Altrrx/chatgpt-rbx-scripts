c/ local activeParts={} local maxParts=350 local function Shoot() if #activeParts>=maxParts then activeParts[1]:Destroy() table.remove(activeParts,1) end local part=Instance.new("Part",workspace) part.Position=owner.Character.Head.Position+owner.Character.Head.CFrame.LookVector*5 local lookvector=owner.Character.Head.CFrame.LookVector*2 part.Size=Vector3.new(1,1,1) part.Anchored=true table.insert(activeParts,part) local projectiledel,shots,maxShots,damageAmount=0,0,10,math.random(5,15) local light=Instance.new("PointLight",part) light.Color=Color3.fromHSV(0,1,1) light.Range=30 light.Brightness=2 part.Touched:Connect(function(hit) local char=hit.Parent if game.Players:GetPlayerFromCharacter(char) then local humanoid=char:FindFirstChildOfClass("Humanoid") if humanoid then humanoid:TakeDamage(damageAmount) end else part:Destroy() end end) game:GetService("Debris"):AddItem(part,5) while wait(0.015) do if shots>=maxShots then break end part.Position=part.Position+lookvector projectiledel=projectiledel+0.015 if projectiledel>=3 then projectiledel=0 shots=shots+1 end end end local tool=Instance.new("Tool",owner.Backpack) tool.Name="funni magik" tool.RequiresHandle=true local handle=Instance.new("Part",tool) handle.Name="Handle" handle.Size=Vector3.new(1,1,1) handle.Anchored=false handle.CanCollide=false handle.Material=Enum.Material.Neon local pointLight=Instance.new("PointLight",handle) pointLight.Color=Color3.fromHSV(0,1,1) pointLight.Range=10 pointLight.Brightness=2 tool.Activated:Connect(Shoot)
